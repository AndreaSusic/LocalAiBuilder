# 1. Scaffold a new React/Vite app under "dashboard"
npx create-vite@latest dashboard -- --template react
cd dashboard
npm install

# 2. Configure Vite to listen on all hosts and allow the Replit domain:
#    Edit vite.config.js to include:
#
#    import { defineConfig } from 'vite'
#    import react from '@vitejs/plugin-react'
#
#    export default defineConfig({
#      plugins: [react()],
#      server: {
#        host: '0.0.0.0',
#        port: Number(process.env.PORT) || 5173,
#        strictPort: true,
#        allowedHosts: 'all',
#        hmr: { protocol: 'wss', clientPort: Number(process.env.PORT) || 5173 }
#      }
#    })

# 3. Install any extras (Headless UI for overlay dialogs):
npm install @headlessui/react
